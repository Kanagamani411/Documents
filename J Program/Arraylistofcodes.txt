How to add objects in arraylist:
Refer https://www.delftstack.com/howto/java/adding-objects-to-arraylist/
class Employee
{
   int id;
   String name;
   
   Employee(int id, String name){
     this.id = id;
     this.name = name;
   }
}
class main
{
  public static void main(String[] args){
    Employee emp1 = new Employee(1, "kax");
    Employee emp2 = new Employee(2, "maha");
    Employee emp3 = new Employee(3, "thendral");
    ArrayList<Employee> arr = new ArrayList<Employee> ();
    arr.add(emp1, emp2, emp3);
    System.out.println(arr);
  }
}
