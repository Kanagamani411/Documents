
public class Logger {

	private static Logger instance;
	private Logger()
	{
	    
	}
	public static synchronized Logger getInstace() {
		if(instance == null) {
			instance = new Logger();
		}
		return instance;
	}
	public synchronized void log(String message) {
		System.out.println(message);
	}
}
